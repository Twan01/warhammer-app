  setSearchText: (text: string) => void;
  setPhaseFilter: (phase: string | null) => void;
  setCpFilter: (cp: string | null) => void;
  clearFilters: () => void;
}

  phaseFilter: null,

  setPhaseFilter: (phase) => set({ phaseFilter: phase }),

}));
