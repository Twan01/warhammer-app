  "Assembly":    "bg-slate-400/30 text-foreground",
  "Painting":    "bg-violet-400/30 text-foreground",
  "Finishing":   "bg-emerald-400/30 text-foreground",
  "Done":        "bg-battle-gold/30 text-foreground",

5. **Replace** the component render body. The `HobbyPipelineProps` interface and function signature stay the same. Replace the render return:

```typescript
export function HobbyPipeline({ units }: HobbyPipelineProps) {
  return (
    <Card className="bg-card border border-border/60 shadow-sm px-6 py-6">
      <p className="text-xs font-semibold uppercase tracking-widest text-muted-foreground mb-4">
        Pipeline
      </p>
      <ol className="flex items-end gap-4" role="list">
        {BUCKET_ORDER.map((bucket) => {
          const count = BUCKET_GROUPS[bucket].reduce(
            (sum, status) => sum + units.filter((u) => u.status_painting === status).length,
            0
          );
          return (
            <li
              key={bucket}
              className="flex flex-1 flex-col items-center gap-1"
              aria-label={`${bucket}: ${count} units`}
            >
              <span className="text-xs text-muted-foreground text-center">
                {bucket}
              </span>
              <span
                className={`inline-flex items-center justify-center min-w-[32px] h-7 px-2 rounded-full text-sm font-semibold tabular-nums ${BUCKET_BUBBLE_CLASS[bucket]}`}
              >
                {count}
              </span>
            </li>
          );
        })}
      </ol>
    </Card>
  );
}
```

Key differences from the old implementation:
- `BUCKET_ORDER.map()` instead of `PAINTING_STATUS_ORDER.map()` — iterates 5 buckets not 11 stages
- `BUCKET_GROUPS[bucket].reduce()` sums counts across all statuses in a bucket
- `aria-label` uses the bucket name, not the individual stage name
- No more `flex-wrap` — 5 wider buckets fill the row without wrapping (use `flex` not `flex-wrap`)
- Each bucket uses `flex-1` so all 5 buckets get equal width
- Bubble labels are the bucket name strings directly (not the STAGE_LABEL_SHORT abbreviations)
