    ```typescript
    function formatSyncDate(iso: string | null): string {
      if (!iso) return "---";
      try {
        return new Date(iso).toLocaleDateString("en-GB", {
          day: "2-digit",
          month: "short",
          year: "numeric",
          hour: "2-digit",
          minute: "2-digit",
        });
      } catch {
        return iso;
      }
    }
    ```
