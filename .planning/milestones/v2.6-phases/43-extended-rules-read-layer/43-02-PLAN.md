const stratagemsByPhase = useMemo(() => {
  const map = new Map<string, RwStratagem[]>();
  for (const s of stratagems) {
    const key = s.phase ?? "Other";
    const group = map.get(key) ?? [];
    group.push(s);
    map.set(key, group);

  return map;
