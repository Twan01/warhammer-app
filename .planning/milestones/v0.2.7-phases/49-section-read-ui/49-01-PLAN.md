    import { useMemo } from "react";
    import type { RecipeSection } from "@/types/recipeSection";

    import type { Paint } from "@/types/paint";

    **Component structure:**

    - Props interface `SectionedTimelineProps`:
      - `sections: RecipeSection[]`
      - `steps: RecipeStep[]`
      - `paintMap: Map<number, Paint>`
      - `stepPhotoUrls?: Map<number, string>`

    - `SectionedTimeline` function component:
      - If `sections.length === 0`, return `null`
      - `stepsBySection = useMemo` grouping steps by `section_id` into `Map<number, RecipeStep[]>`. Steps with `section_id === null` are skipped.
      - `sectionAvailability = useMemo` computing `Map<number, { owned: number; missing: number }>` per section. For each step with `paint_id !== 0`, look up paint in `paintMap`, call `isPaintMissing(paint)`. Count owned vs missing per `section_id`.
      - Render outer div `className="flex flex-col gap-6"` with `data-testid="sectioned-timeline"`
      - Map over `sections`, for each section:
        - Compute `sectionSteps = stepsBySection.get(section.id) ?? []`
        - Compute `totalMinutes`: if `sectionSteps.every(s => s.time_estimate_minutes === null)` then `null`, else `sectionSteps.reduce((sum, s) => sum + (s.time_estimate_minutes ?? 0), 0)`
        - Compute `availability = sectionAvailability.get(section.id)`
        - Render section header div `className="flex items-center gap-2"` with `data-testid="section-header"`:
          - `<span className="text-sm font-semibold">{section.name}</span>`
          - If `section.surface`: `<Badge variant="outline" className="text-xs">{section.surface}</Badge>`
          - If `section.optional === 1`: `<Badge variant="outline" className="text-xs">Optional</Badge>`
          - Right-side metadata `className="text-xs text-muted-foreground ml-auto flex items-center gap-3"`:
            - Step count: `<span className="flex items-center gap-0.5"><Layers className="h-3 w-3" />{stepCount} {stepCount === 1 ? "step" : "steps"}</span>`
            - If `totalMinutes !== null`: `<span className="flex items-center gap-0.5"><Clock className="h-3 w-3" />{totalMinutes} min</span>`
            - If `availability` exists and `(availability.owned > 0 || availability.missing > 0)`:
              - Render availability inline:
                - Green dot `<span className="inline-block h-2 w-2 rounded-full shrink-0" style={{ backgroundColor: "#22c55e" }} />` + `{availability.owned} owned`
                - If `availability.missing > 0`: separator `<span className="mx-0.5">·</span>` + red dot `style={{ backgroundColor: "#ef4444" }}` + `{availability.missing} missing`
        - Render `<RecipeStepTimeline steps={sectionSteps} paintMap={paintMap} stepPhotoUrls={stepPhotoUrls} />` below the header

    **Export:** Named export `SectionedTimeline` and `SectionedTimelineProps`.
