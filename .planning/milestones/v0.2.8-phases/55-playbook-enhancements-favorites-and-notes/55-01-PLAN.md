interface RuleNoteEditorProps {

}

Implementation:
- Local state: `const [localText, setLocalText] = useState(note?.note_text ?? "")`.

- Cleanup effect: `useEffect(() => () => { if (timerRef.current) clearTimeout(timerRef.current); }, []);`
- Render: `<Separator className="my-2" />` then `<p className={SECTION_LABEL_CLASS}>Your Notes</p>` then `<textarea className={TEXTAREA_CLASS + " min-h-[56px]"} rows={2} value={localText} onChange={handleChange} placeholder="Add personal notes..." />`. Import `SECTION_LABEL_CLASS = "text-xs font-semibold text-muted-foreground uppercase tracking-wide"` and `TEXTAREA_CLASS = "flex min-h-[80px] w-full rounded-md border border-input bg-background px-3 py-2 text-sm ring-offset-background placeholder:text-muted-foreground focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 disabled:cursor-not-allowed disabled:opacity-50"` — define these as local constants (same values as PlaybookTab).
